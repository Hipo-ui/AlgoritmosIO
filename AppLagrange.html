        % Button pushed function: CalcularButton
        function CalcularButtonPushed(app, event)
            exp = app.ExponenteEditField.Value;
            exp2 = app.Exponente2EditField.Value;
            indice = app.IndiceEditField.Value;
            presupuesto = app.PresupuestoEditField.Value;
            cTonelada = app.CostoToneladaEditField.Value;
            cHora = app.CostoxHoraEditField.Value;

            syms s h l

            f = (indice*h^exp)*(s^exp2) - l*cHora*h - l*cTonelada*s + l*presupuesto
            fh = diff(f,h)
            fs = diff(f,s)
            fl = diff(f,l)

            eq1 = fh == 0
            eq2 = fs == 0
            eq3 = fl == 0

            hola = solve([eq1, eq2, eq3])
            respuestaH = eval(hola.h)
            respuestaS = eval(hola.s)
            respuestaR = indice*(respuestaH^exp)*(respuestaS^exp2)
            app.NmerodeHorasEditField.Value = respuestaH
            app.NmerodeToneladasEditField.Value = respuestaS
            app.MximodeIngresosEditField.Value = respuestaR
        end
    end
